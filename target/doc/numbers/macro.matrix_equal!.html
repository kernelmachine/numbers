macro_rules! matrix_equal {
    ( $x:expr , $y : expr) => (
        {
            assert!($x.row_size == $y.row_size);
            assert!($x.col_size == $y.col_size);
            assert!($x.transpose  == $y.transpose);
            if let Ok(s) = &$x - &$y{
                for elem in s.elements{
                    assert!(elem.abs() < 1e14)
                };
                return Ok(true)
            }
            Ok(false)
        }
